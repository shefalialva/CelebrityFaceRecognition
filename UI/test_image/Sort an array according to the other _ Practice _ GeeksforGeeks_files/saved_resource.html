class Solution{
    public:

//a comparator function which tries to sort the elemets based on their priority stored in map or if the priorities are same, it compares element's value and sorts them in ascending order.
    static bool comp(pair<int,int> a, pair<int,int> b){
           if(a.second<b.second)
             return true;
             if(a.second==b.second)
              return a.first<b.first;
             return false;
            
    }
   
    vector<int> sortA1ByA2(vector<int> A1, int N, vector<int> A2, int M) 
    {

//here map stores the priority of the elements in A2
          map<int,int>mp;
          mp.clear();
          for(int i=0; i<M; i++){
                if(mp[A2[i]]==0 )
                   mp[A2[i]]=i+1;
          }
        vector<pair<int,int>>vp;
        for(int i=0; i<N; i++){
             if(mp[A1[i]]==0 )
              vp.push_back({A1[i],N});
              else
              vp.push_back({A1[i],mp[A1[i]]});
        }
        sort(vp.begin(),vp.end(),comp);
        for(int i=0; i<N; i++){
               A1[i]=vp[i].first;
        }
        return A1;
    } 
};

1. create a map and store frequencies for elementsin array1 present in array2 as well

2. Traverse array1 and if element not in the map push to last

3. Add the elements*frequency times to the output array

4. Add the remaining elements to the array

5. Sort the remaining elements part

 

    vector<int> sortA1ByA2(vector<int> a, int n, vector<int> b, int m) 
    {
        unordered_map<int,int> mp;
        vector<int> ans(n);
        for(int i=0;i<m;i++)
            mp[b[i]] = 1;

            
        for(int i=0;i<n;i++) {
            if(mp.find(a[i]) != mp.end())
                mp[a[i]]++;
        }

        int end= n-1;
        int start=0;
        while(start <= end) {
            if(mp.find(a[start]) == mp.end()) {
                swap(a[start], a[end]);
                end--;
            }
                
            else
                start++;
        }
        
        int k=0;
        for(int i=0;i<m;i++) {
            for(int j=1;j<mp[b[i]];j++)
                ans[k++] = b[i];
        }
        
        for(int i=start;i<n;i++)
            ans[k++] = a[i];
            
        sort(ans.begin() + start, ans.end());
        
        return ans;
    }

vector<int> sortA1ByA2(vector<int> A1, int N, vector<int> A2, int M) 
    {
        map<int,int>mp;
        vector<int>ans;
        for(auto x:A1){
            mp[x]++;
        }
        for(auto temp:A2){
            if(mp.find(temp)!=mp.end()){
                auto x=mp.find(temp);
                int count=x->second;
                vector<int>v(count,temp);
                ans.insert(ans.end(),v.begin(),v.end());
                mp.erase(temp);
            }
        }
        for(auto x:mp){
            int element=x.first;
            int count=x.second;
            vector<int>v(count,element);
            ans.insert(ans.end(),v.begin(),v.end());
        }
        return ans;
    }

    vector<int> sortA1ByA2(vector<int> A1, int N, vector<int> A2, int M) 
    {
        unordered_map<int,int> mp;
        for(int x: A1) 
            mp[x]++;
        
        int w = 0;
        for(int x : A2) {
            while(mp[x]-- ) A1[w++] = x;
            mp.erase(x);
        }
        int s = w;
        for(auto &x: mp){
            while(x.second--) 
                A1[w++] = x.first;
        }
        sort(begin(A1)+s, end(A1));
        return A1;
    } 
};
